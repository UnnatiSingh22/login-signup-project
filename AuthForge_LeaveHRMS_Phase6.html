SQL JOIN Concepts Used in This System

Every employee record you see is built with these JOIN patterns (simulated in JS/localStorage)

INNER JOIN (One-to-Many)
SELECT e.*, d.name AS dept
FROM employees e
INNER JOIN departments d
ON e.dept_id = d.id
JOIN Bridge (Many-to-Many)
SELECT e.name, s.skill
FROM employees e
JOIN emp_skills es ON e.id=es.emp_id
JOIN skills s ON es.skill_id=s.id
LEFT JOIN (Aggregation)
SELECT d.name,
COUNT(e.id) AS headcount
FROM departments d
LEFT JOIN employees e ON d.id=e.dept_id
GROUP BY d.id

Protected Route Middleware (Node.js concept)

        
// middleware/auth.js
const jwt = require('jsonwebtoken');

function protectRoute(req, res, next) {
  const token = req.headers['authorization']?.split(' ')[1];
  if (!token) return res.status(401).json({ error: 'No token' });
  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.user = decoded;  // { sub, email, role }
    next();
  } catch {
    return res.status(403).json({ error: 'Token invalid or expired' });
  }
}

function adminOnly(req, res, next) {
  if (req.user?.role !== 'admin') return res.status(403).json({ error: 'Admin access required' });
  next();
}

BEGIN TRANSACTION
BEGIN;
UPDATE leave_requests
 SET status='approved'
 WHERE id=$1;
UPDATE leave_balance
 SET used=used+$days
 WHERE emp_id=$2;

          
INSERT AUDIT LOG
INSERT INTO audit_logs
(leave_id,acted_by,
 action,comment,ts)
VALUES
($1,$2,$3,$4,NOW());

          
COMMIT / ROLLBACK
COMMIT;
-- on error:
ROLLBACK;
-- Atomicity ensures
-- balance+status
-- always in sync

// controllers/assetController.js
const AssetService = require('../services/assetService');

exports.getAllAssets = async (req, res, next) => {
  try {
    const { page=1, limit=10, type, status, search } = req.query;
    const result = await AssetService.getAll({ page, limit, type, status, search });
    res.json({ success: true, data: result.data, meta: result.meta });
  } catch (err) { next(err); }
};

exports.assignAsset = async (req, res, next) => {
  try {
    const { assetId, empId, notes } = req.body;
    const asset = await AssetService.assign(assetId, empId, notes, req.user);
    res.json({ success: true, data: asset });
  } catch (err) { next(err); }
};

        
 Service Layer
Business logic, validation, orchestration
// services/assetService.js
const AssetRepo = require('../repositories/assetRepo');
const AuditService = require('./auditService');
const NotifService = require('./notifService');

exports.assign = async (assetId, empId, notes, actingUser) => {
  const asset = await AssetRepo.findById(assetId);
  if (!asset) throw new AppError('Asset not found', 404);
  if (asset.status !== 'available')
    throw new AppError('Asset not available', 400);
  const updated = await AssetRepo.assign(assetId, empId, notes);
  await AuditService.log('asset','ASSIGN',assetId, {status:'available'}, {status:'assigned',empId}, actingUser);
  await NotifService.send(empId, 'asset', `Asset "${asset.name}" assigned to you`);
  return updated;
};

        
 Repository Layer
Data access — only layer that touches the DB
// repositories/assetRepo.js
const { pool } = require('../db');

exports.findById = async (id) => {
  const { rows } = await pool.query(
    'SELECT * FROM assets WHERE id = $1', [id]
  );
  return rows[0] || null;
};

exports.assign = async (assetId, empId, notes) => {
  const { rows } = await pool.query(`
    UPDATE assets
    SET status='assigned', assigned_to=$1,
        assigned_at=NOW(), notes=$2
    WHERE id=$3 RETURNING *`, [empId, notes, assetId]
  );
  return rows[0];
};

        
 Validation & Error Middleware
Input validation + centralized error handling
// middleware/validate.js
const Joi = require('joi');

const assignSchema = Joi.object({
  assetId: Joi.string().uuid().required(),
  empId: Joi.string().uuid().required(),
  notes: Joi.string().max(500)
});

exports.validateAssign = (req, res, next) => {
  const { error } = assignSchema.validate(req.body);
  if (error) return res.status(400).json({ error: error.details[0].message });
  next();
};

// middleware/errorHandler.js
module.exports = (err, req, res, next) => {
  const status = err.statusCode || 500;
  res.status(status).json({ success:false, error: err.message });
};

-- Employee Full Profile View
CREATE VIEW v_employee_profile AS
SELECT
  e.id, e.fname, e.lname, e.email,
  e.position, e.salary, e.status,
  d.name AS department,
  d.head AS dept_head,
  lb.casual_remaining,
  lb.sick_remaining,
  COUNT(a.id) FILTER (WHERE a.status='assigned') AS assets_count,
  COUNT(lr.id) FILTER (WHERE lr.status='approved') AS leaves_taken
FROM employees e
LEFT JOIN departments d ON d.id = e.dept_id
LEFT JOIN leave_balance_view lb ON lb.emp_id = e.id
LEFT JOIN assets a ON a.assigned_to = e.id
LEFT JOIN leave_requests lr ON lr.emp_id = e.id
GROUP BY e.id, d.name, d.head, lb.casual_remaining, lb.sick_remaining;

-- Asset Utilization View
CREATE VIEW v_asset_utilization AS
SELECT
  type,
  COUNT(*) AS total,
  COUNT(*) FILTER (WHERE status='assigned') AS in_use,
  COUNT(*) FILTER (WHERE status='available') AS available,
  ROUND(COUNT(*) FILTER (WHERE status='assigned') * 100.0 / COUNT(*), 1) AS utilization_pct
FROM assets
GROUP BY type;

        
 Stored Procedures
-- Atomic Leave Approval Procedure
CREATE OR REPLACE PROCEDURE approve_leave(
  p_leave_id UUID,
  p_actor_id UUID,
  p_role TEXT,
  p_comment TEXT
)
LANGUAGE plpgsql AS $$
DECLARE v_emp_id UUID; v_type TEXT; v_days INT;
BEGIN
  SELECT emp_id, type, days INTO v_emp_id, v_type, v_days
  FROM leave_requests WHERE id = p_leave_id;

  UPDATE leave_requests
  SET status = 'approved', hr_comment = p_comment,
      updated_at = NOW()
  WHERE id = p_leave_id;

  UPDATE leave_balance
  SET used = used + v_days
  WHERE emp_id = v_emp_id AND type = v_type
    AND year = EXTRACT(YEAR FROM NOW());

  INSERT INTO audit_logs(module, action, record_id, acted_by, new_val)
  VALUES ('leave','APPROVE',p_leave_id,p_actor_id,
    jsonb_build_object('status','approved','days',v_days));

  COMMIT;
EXCEPTION WHEN OTHERS THEN ROLLBACK; RAISE;
END; $$;

CREATE TABLE assets (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,

  status TEXT DEFAULT 'available',

 Pagination + Filtering + Sorting
-- Dynamic query with pagination, filter & sort
-- repositories/assetRepo.js
exports.getAll = async ({ page=1, limit=10, type, status, search, sortBy='created_at', order='DESC' }) => {
  const offset = (page-1) * limit;
  const conditions = ['1=1'];
  const params = [];
  let i = 1;

  if (type)   { conditions.push(`type = $${i++}`);   params.push(type); }
  if (status) { conditions.push(`status = $${i++}`); params.push(status); }
  if (search) { conditions.push(`name ILIKE $${i++}`); params.push(`%${search}%`); }

  const allowed = ['name','type','status','created_at'];
  const col = allowed.includes(sortBy) ? sortBy : 'created_at';
  const dir = order==='ASC' ? 'ASC' : 'DESC';

  const countQ = `SELECT COUNT(*) FROM assets WHERE ${conditions.join(' AND ')}`;
  const dataQ  = `SELECT * FROM assets WHERE ${conditions.join(' AND ')}
                  ORDER BY ${col} ${dir} LIMIT $${i} OFFSET $${i+1}`;
  params.push(limit, offset);

  const [count, rows] = await Promise.all([
    pool.query(countQ, params.slice(0,-2)),
    pool.query(dataQ, params)
  ]);
  return { data: rows.rows, meta: { total: +count.rows[0].count, page, limit, pages: Math.ceil(+count.rows[0].count/limit) } };
};
