SQL JOIN Concepts Used in This System

Every employee record you see is built with these JOIN patterns (simulated in JS/localStorage)

INNER JOIN (One-to-Many)
SELECT e.*, d.name AS dept
FROM employees e
INNER JOIN departments d
ON e.dept_id = d.id
JOIN Bridge (Many-to-Many)
SELECT e.name, s.skill
FROM employees e
JOIN emp_skills es ON e.id=es.emp_id
JOIN skills s ON es.skill_id=s.id
LEFT JOIN (Aggregation)
SELECT d.name,
COUNT(e.id) AS headcount
FROM departments d
LEFT JOIN employees e ON d.id=e.dept_id
GROUP BY d.id

Protected Route Middleware (Node.js concept)

        
// middleware/auth.js
const jwt = require('jsonwebtoken');

function protectRoute(req, res, next) {
  const token = req.headers['authorization']?.split(' ')[1];
  if (!token) return res.status(401).json({ error: 'No token' });
  try {
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    req.user = decoded;  // { sub, email, role }
    next();
  } catch {
    return res.status(403).json({ error: 'Token invalid or expired' });
  }
}

function adminOnly(req, res, next) {
  if (req.user?.role !== 'admin') return res.status(403).json({ error: 'Admin access required' });
  next();
}

PostgreSQL Transaction Pattern

        
Atomic update: leave status + balance update in a single transaction

        

          
BEGIN TRANSACTION
BEGIN;
UPDATE leave_requests
 SET status='approved'
 WHERE id=$1;
UPDATE leave_balance
 SET used=used+$days
 WHERE emp_id=$2;

          
INSERT AUDIT LOG
INSERT INTO audit_logs
(leave_id,acted_by,
 action,comment,ts)
VALUES
($1,$2,$3,$4,NOW());

          
COMMIT / ROLLBACK
COMMIT;
-- on error:
ROLLBACK;
-- Atomicity ensures
-- balance+status
-- always in sync
